Regex使用

exec方法
  • exec选项为g,需要反复执行迭代才能匹配下一个，返回数组
  • exec通配符(*)，返回数组
  • test返回true或false
        var str = "abc abbc abde adccc";
        var regexp = new RegExp('ab',"g");
        var regexp2 = new RegExp('a(b*)b',"g");
        var regexpResult = regexp.exec(str);
        if (regexpResult != null) {
            //["ab", index: 0, input: "abc abbc abde adccc", groups: undefined] 
            console.log(regexpResult); 

            regexpResult=regexp.exec(str)  
            //["ab", index: 4, input: "abc abbc abde adccc", groups: undefined]
            console.log(regexpResult);  
        }
        else {
            console.log('未找到匹配项');
        }
        //["ab", "", index: 0, input: "abc abbc abde adccc", groups: undefined]
        console.log('regex2',regexp2.exec(str));
        //["abb", "b", index: 4, input: "abc abbc abde adccc", groups: undefined]
        console.log('regex2',regexp2.exec(str));
