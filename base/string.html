string字符串的方法

  • match匹配
  • search查询
  • repalce替换
  • split切割
   
            var str = "asbc   abbc  abde    adccc";    
            // ["ab", "ab"]
            console.log(str.match(/ab/g));
            //5
            console.log(str.search(/ab*c/));
            //asb9 abb9 abde ad999
            console.log(str.replace(/c/g,'9')); 
            //\s代表空格,按一个或多个空格来分割 
            // ["asbc", "abbc", "abde", "adccc"]
            console.log(str.split(/\s+/));
